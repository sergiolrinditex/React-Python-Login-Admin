## I. Implementation plan (developer)

### I.1 — `scripts/gen-dev-secrets.sh` design (NEW, ~80-120 LOC)

```
#!/usr/bin/env bash
# Idempotent dev-only secrets provisioner for .env.
#
# Slice:  P01-S02-T009 — generate ≥32-byte JWT dev key + default
#         ENABLE_VERBOSE_LOGGING in dev .env
# Phase:  P01 Auth + Data Foundation
# Promoted from FU-20260512044309 (origin P01-S02-T002 verify-slice).
#
# Responsibility (single):
#   Ensure the developer-local .env file ships RFC-7518-compliant JWT key
#   and verbose logging enabled BY DEFAULT, without ever overwriting a
#   real human-set value.
#
# Behaviors:
#   1. If .env does not exist, copy .env.example to .env first.
#   2. JWT_PRIVATE_KEY:
#        a. Read current value.
#        b. If value matches placeholder pattern ('replace-with-dev-key',
#           empty, or len<32 bytes) → generate a new
#           secrets.token_urlsafe(48) key (~64 url-safe chars, >=32 bytes)
#           and replace.
#        c. Else → leave untouched (developer set a real key).
#      Also do this for JWT_PUBLIC_KEY for HS256 symmetry consistency
#      (decoder uses _JWT_KEY = JWT_PRIVATE_KEY, but TECHNICAL_GUIDE §11.1
#       declares both — keep .env honest).
#   3. ENABLE_VERBOSE_LOGGING:
#        a. If line exists → set value to 'true' (overwrite false→true is OK,
#           it's the dev default per acceptance).
#        b. If line missing → append.
#   4. Logging:
#        - BEFORE: 'gen-dev-secrets.start path=$ENV_PATH'
#        - per change: 'gen-dev-secrets.jwt_private_key.rotated len=64 reason=placeholder'
#                      'gen-dev-secrets.jwt_private_key.kept len=64 reason=already_real'
#                      'gen-dev-secrets.enable_verbose_logging.set to=true was=false'
#        - AFTER: 'gen-dev-secrets.done changed=N'
#        - ERROR: 'gen-dev-secrets.error context=...'
#        - NEVER print the actual key value. Only its length.
#   5. Exit 0 always (no-op-on-no-change is success).
#
# Implementation details:
#   - POSIX bash, no GNU-only flags (no `sed -i`).
#   - Atomic write: build a new file at $ENV_PATH.tmp, mv to $ENV_PATH.
#   - Detection of placeholder: regex '^(replace-with-dev-key|)$' OR length<32.
#   - Generation: python3 -c 'import secrets; print(secrets.token_urlsafe(48))'.
#     ~64 url-safe chars; >=32 bytes; cryptographically strong.
#
# Re-run safety: idempotent. After first run, all subsequent runs are no-op.

set -euo pipefail

ROOT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")/.." && pwd)"
ENV_PATH="${ROOT_DIR}/.env"
ENV_EXAMPLE="${ROOT_DIR}/.env.example"

log() { printf '==> gen-dev-secrets.%s\n' "$1" >&2; }

# --- 1. Ensure .env exists --------------------------------------------------
if [ ! -f "$ENV_PATH" ]; then
  if [ ! -f "$ENV_EXAMPLE" ]; then
    log "error context=missing_env_example path=$ENV_EXAMPLE"
    exit 1
  fi
  cp "$ENV_EXAMPLE" "$ENV_PATH"
  log "env_created from=$ENV_EXAMPLE to=$ENV_PATH"
fi

log "start path=$ENV_PATH"

CHANGED=0

# --- 2. Helpers -------------------------------------------------------------
gen_jwt_key() {
  python3 -c 'import secrets; print(secrets.token_urlsafe(48))'
}

current_value() {
  # current_value VAR -> echoes value or empty string
  awk -v k="$1" -F= '$1==k { sub(/^[^=]*=/,""); print; exit }' "$ENV_PATH"
}

is_placeholder_key() {
  # is_placeholder_key VALUE -> exit 0 if placeholder (needs rotation)
  local v="$1"
  if [ -z "$v" ] || [ "$v" = "replace-with-dev-key" ]; then return 0; fi
  if [ "${#v}" -lt 32 ]; then return 0; fi
  return 1
}

set_or_append() {
  # set_or_append VAR VALUE -> rewrite .env atomically
  local var="$1" val="$2" tmp
  tmp="$(mktemp "${ENV_PATH}.tmp.XXXXXX")"
  if grep -qE "^${var}=" "$ENV_PATH"; then
    # Replace line
    awk -v k="$var" -v v="$val" -F= 'BEGIN{OFS="="} $1==k {print k"="v; next} {print}' \
      "$ENV_PATH" > "$tmp"
  else
    # Append line
    cat "$ENV_PATH" > "$tmp"
    printf '%s=%s\n' "$var" "$val" >> "$tmp"
  fi
  mv "$tmp" "$ENV_PATH"
}

# --- 3. JWT_PRIVATE_KEY -----------------------------------------------------
JWT_CUR="$(current_value JWT_PRIVATE_KEY)"
if is_placeholder_key "$JWT_CUR"; then
  NEW_KEY="$(gen_jwt_key)"
  set_or_append JWT_PRIVATE_KEY "$NEW_KEY"
  set_or_append JWT_PUBLIC_KEY  "$NEW_KEY"  # HS256 symmetric: same key for verify
  log "jwt_private_key.rotated len=${#NEW_KEY} reason=placeholder"
  log "jwt_public_key.rotated len=${#NEW_KEY} reason=hs256_symmetric"
  CHANGED=$((CHANGED + 2))
  unset NEW_KEY  # drop from env immediately
else
  log "jwt_private_key.kept len=${#JWT_CUR} reason=already_real"
fi

# --- 4. ENABLE_VERBOSE_LOGGING ---------------------------------------------
EVL_CUR="$(current_value ENABLE_VERBOSE_LOGGING)"
if [ "$EVL_CUR" != "true" ]; then
  set_or_append ENABLE_VERBOSE_LOGGING "true"
  log "enable_verbose_logging.set to=true was=${EVL_CUR:-<missing>}"
  CHANGED=$((CHANGED + 1))
else
  log "enable_verbose_logging.kept to=true"
fi

log "done changed=$CHANGED"
exit 0
```
